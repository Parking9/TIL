```python
    if direc == 1:
        lst=[lst[-1]]+lst[:-1]

        lst=lst[1:]+[lst[0]]
```

## 7-bit 10진수

```python
for tc in range(1,int(input())+1):
    ipt=input()
    cnt=0
    ans=[]
    while cnt !=10:
        st = ipt[cnt*7:(cnt+1)*7]
        ans.append(int(st,2))
        cnt+=1
    print(f'#{tc}', end=' ')
    print(*ans, end=' ')
    print()
```
