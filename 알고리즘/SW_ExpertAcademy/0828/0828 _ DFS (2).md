## 1979. 단어가 어디에 들어갈 수 있을까 (2 가지)

```python
# 어디에 단어가 들어갈 수 있을까
T = int(input())
for tc in range(1, T+1):
    N, K = map(int, input().split())
    arr = [list(map(int, input().split())) for _ in range(N)]
    ans = 0
    for i in arr:
        v = ''
        for j in i:
            v += str(j)
        for k in v.split('0'):
            if len(k) == K:
                ans += 1

    for i2 in range(N):
        v2 = ''
        for j2 in range(N):
            v2 += str(arr[j2][i2])
        for k2 in v2.split('0'):
            if len(k2) == K:
                ans += 1
    print(f'#{tc} {ans}')
```

## 4615. 재미있는 오셀로 게임

```python
# 오셀로
# 좌표의 범위 에러 체크 함수
def check(x, y, N):
    if x >= 0 and x < N and y >= 0 and y < N:
        return True
    else:
        return False

# 돌을 놓으면 변경된 arr를 출력하는 함수


def othello(x, y, N, c, arr):
    # 8가지 방향 중 범위를 벗어나지 않고 다른 색을 가진 방향 리스트 생성
    v = []
    for i in range(-1, 2):
        for j in range(-1, 2):
            if check(x+i, y+j, N) and (arr[x+i][y+j] not in [c, 0]):
                v.append([i, j])
    # 범위 후보 중 나아갔을때 같은 색이 있는 방향만 선택
    ans = []
    for dir in v:
        x2 = x
        y2 = y
        cnt = 1
        while check(x2+dir[0]*cnt, y2+dir[1]*cnt, N):
            if arr[x2+dir[0]*cnt][y2+dir[1]*cnt] == c:
                ans.append(dir)
                break
            else:
                cnt += 1
    # 최종 선택된 방향들을 나아가며 같은 색이 나오기 전까지 색칠
    for dir2 in ans:
        cnt = 1
        while arr[x+dir2[0]*cnt][y+dir2[1]*cnt] != c:
            arr[x+cnt*dir2[0]][y+dir2[1]*cnt] = c
            cnt += 1
    # arr 반환
    return arr


T = int(input())
for tc in range(1, T+1):
    N, M = map(int, input().split())
    arr = [[0]*N for _ in range(N)]
    arr[N//2][N//2] = 'W'
    arr[(N//2)-1][N//2] = 'B'
    arr[(N//2)-1][(N//2)-1] = 'W'
    arr[N//2][(N//2)-1] = 'B'

    for tn in range(M):
        x, y, c = map(int, input().split())
        x -= 1
        y -= 1
        if c == 1:
            c = 'B'
        else:
            c = 'W'
        arr[x][y] = c
        arr = othello(x, y, N, c, arr)

    numb1, numb2 = 0, 0
    for r in arr:
        for r2 in r:
            if r2 == 'W':
                numb2 += 1
            elif r2 == 'B':
                numb1 += 1
    print(f'#{tc} {numb1} {numb2}')

```
